# switch to different branch
git checkout branch_name

# git branch lists all branches and has * next to the branch we are in

# merge branch
git merge source desitnation
git merge branch_name main

git clone path_to_project_directory repo_name

git pull origin main

git remote

# local changes must be commited first
git push remote local_branch
